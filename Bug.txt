using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace WindowsFormsApp1
{
   public class Bug: Animal
    {
       
        public t tip;/*Тип*/
        public Bug()/*Конструктор класса*/
        {
            tip = Bug.t.NULL;
            set_Meal(10);
        }
        
  
        public int check_Meal(Soil[,] arr, int a, int b)
        {
            if (get_Meal() <= 0 && tip==Bug.t.SELF)/*Если уровень насыщенности равен 0, то животное умирает*/
            {
                tip = Bug.t.NULL;
                arr[a, b].cell = Soil.cells.SELF;
                return -1;
            }
           else if (get_Meal() >= 12)/*Если уровень насыщенности больше или равен максимуму, то уровень насыщенности выставляется по максимуму*/
            {
                set_Meal(12);
                return 0;
            }
            return 0;
        }

     
    }
}
