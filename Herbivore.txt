using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace WindowsFormsApp1
{
    public class Herbivore: Animal
    {
       
        public t tip;/*Тип*/
        public Herbivore()/*Конструктор класса*/
        {
            tip = Herbivore.t.NULL;
            set_Meal(15);
        }
      

       
        public int check_Meal(Soil[,] arr, int a, int b)
        {
            if (get_Meal() <= 0 && tip==Herbivore.t.SELF)/*Если уровень насыщенности равен 0, то животное умирает*/
            {
                tip = Herbivore.t.NULL;
                arr[a, b].cell = Soil.cells.SELF;
                return -1;
            }
           else if (get_Meal() >= 16 && tip == Herbivore.t.SELF)/*Если уровень насыщенности больше или равен максимуму, то уровень насыщенности выставляется по максимуму*/
            {
                set_Meal(16); 
            }
            return 0;
        }

      
    }
}
